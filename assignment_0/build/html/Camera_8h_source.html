56  void init()
57  {
58  // compute viewing direction and distance of eye to scene center
59  vec3 view = normalize(center - eye);
60  double dist = distance(center, eye);
61 
62  // compute width & height of the image plane
63  // based on the opening angle of the camera (fovy) and the distance
64  // of the eye to the near plane (dist)
65  double w = width;
66  double h = height;
67  double image_height = 2.0 * dist * tan(0.5*fovy/180.0*M_PI);
68  double image_width = w/h * image_height;
69 
70  // compute right and up vectors on the image plane
71  x_dir = normalize( cross(view, up) ) * image_width / w;
72  y_dir = normalize( cross(x_dir, view) ) * image_height / h;
73 
74  // compute lower left corner on the image plane
75  lower_left = center - 0.5f*w*x_dir - 0.5f*h*y_dir;
76  }
